图片的自动轮播

需求:自定义一组图片，每2秒进行图片切换，最终的结果是一直的切换不停

  • 事件:setInterval()
  • 事件源:图片
  • 响应行为:不停的切换图片

文档加载事件:window.onload()

    <script>
            var index = 0;
            /*图片切换的方法*/
            function changeImg(){
                var img1 = document.getElementById("img1");
                //计算下切换图片的张数
                var curIndex = index%3+1;
                img1.src = "../img/"+curIndex+".jpg";
                //每次切换图片后进行索引+1
                index = index+1;
            }

            //定时器的启动方法
            function init(){
                setInterval("changeImg()",1000);
            }

    </script>
    </head>
    <body onload="init()">
        <img src="../img/1.jpg" width="100%" id="img1" />
    </body>

广告的定时弹出与隐藏

定义一个图片，当我们打开网页的时候，5秒钟之后弹出来，显示广告，5秒钟之后，继续隐藏(可以选择手动隐藏或者自动隐藏)

  • 事件:setTimeout
  • 事件源:图片

  • 响应行为:5秒钟之后显示，5秒之后关闭

    <head>
    <meta charset="UTF-8">
    <title></title>
    <script>
        /*显示广告的函数*/
        function showAD(){
            var img1 = document.getElementById("img1");
            //将广告元素显示出来
            img1.style.display = "block";
            //调用隐藏广告的函数
            setTimeout("hideAD()",5000);

        }
        /*隐藏广告的函数*/
        function hideAD(){
            var img1 = document.getElementById("img1");
            //将广告元素隐藏
            img1.style.display = "none";

        }

        //初始化启动的函数
        function init(){
            //调用显示广告的函数
            setTimeout("showAD()",5000);
        }
    </script>
</head>
<body onload="init()">
    <img src="../img/1.jpg" style="display: none;" id="img1" />
</body>

                var row = rows[i];
                if(i%2 == 0){
                    row.bgColor = "red";
